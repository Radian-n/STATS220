Pooh Bayes

library(magick)


# Get pooh-bear meme images
normal_pooh <- image_read("https://i.kym-cdn.com/photos/images/newsfeed/001/474/942/012.gif") %>%
  image_scale(450) %>%
  image_crop("404x400+50") %>%  # Making the image square (original was rectangular)
  image_scale(450)
fancy_pooh <- image_read("https://i.kym-cdn.com/photos/images/newsfeed/001/474/943/b12.jpg") %>%
  image_scale(450)


# Generate Captions
normal_pooh_caption <- image_blank(width = 600, height = 450, color = "#FFFFFF") %>%
  image_annotate(text = "\"Beliefs\"", size = 74, font = "Impact", gravity = "center")
fancy_pooh_caption <- image_blank(width = 600, height = 450, color = "#FFFFFF") %>%
  image_annotate(text = "\"Bayesian Priors\"", size = 74, font = "Impact", gravity = "center")
black_line <- image_blank(width = 450+600, height = 8, color = "#000000")


# Combine Quadrants
top_line.v <- c(normal_pooh, normal_pooh_caption)
top_line.img <- image_append(top_line.v)
bottom_line.v <- c(fancy_pooh, fancy_pooh_caption)
bottom_line.img <- image_append(bottom_line.v)
final_meme.img <- image_append(c(top_line.img, black_line, bottom_line.img), stack = TRUE)


# Preview Meme
final_meme.img


# Save meme to drive
image_write(final_meme.img, "poohbayes_meme.png")

My animated GIF

Not a meme, just counting up from 1 to 4 in gif form with pretty pastel colours.

Pastel counting Gif

library(magick)


# Creating empty vector for images
images2.v = c()
colours.v = c("#dbffcb", "#d3f7f7", "#f8e6dc", "#e8dcf8", "#d3f7f7")  # Repeated colour added because i+1 later

# Creating a gif that counts the frames
for (i in 1: 4) {
  image2 = image_blank(width = 400, height = 400, color = colours.v[i]) %>%
    image_annotate(text = i, size=250, font = "serif", color = colours.v[i+1], gravity = "center")
  images2.v = append(images2.v, image2)
}

# Create gif from generated image vector
count_gif = image_animate(images2.v, delay = 50)

# Write gif to file
image_write(count_gif, "my_animation.gif")
